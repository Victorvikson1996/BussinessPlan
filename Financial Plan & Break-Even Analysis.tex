\begin{itemize}

    \item \textbf{Contribution Margin/Charger:} €48k revenue – €24k variable costs = €24k.
    \item \textbf{Break-Even Point:} €1.2M ÷ €24k = 50 chargers (33\% utilization).
\end{itemize}
